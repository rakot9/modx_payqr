    cartTpl         на shopCart_small

    orderFormPageId на 6
    className       ShopContent
    packageName     shop

* cart_order_page -> Shopkeeper3
    cartRowTpl      на shopCartRow_full
    cartTpl         на shopCart_full
    orderFormPageId на 6
    flyToCart       на nofly
    className       ShopContent
    packageName     shop
